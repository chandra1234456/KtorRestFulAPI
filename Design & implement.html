3. Android Implementation (Kotlin)

3.1 ViewModel Example


class ExpenseViewModel(
    private val repository: ExpenseRepository
) : ViewModel() {

    private val _balances = MutableStateFlow>(emptyMap())
    val balances = _balances.asStateFlow()

    fun loadBalances(groupId: String) {
        viewModelScope.launch(Dispatchers.IO) {
            val result = repository.getBalances(groupId)
            _balances.value = result
        }
    }
}

Interview Explanation:

  • viewModelScope ensures lifecycle safety
  • Dispatchers.IO is used for network calls
  • StateFlow provides observable state to UI

6. Balance Calculation Logic (Core Feature)


fun calculateBalances(expenses: List): Map {
    val balances = mutableMapOf()

    expenses.forEach { expense ->
        val share = expense.amount / expense.participants.size

        expense.participants.forEach { userId ->
            balances[userId] = balances.getOrDefault(userId, 0.0) - share
        }

        balances[expense.paidBy] =
            balances.getOrDefault(expense.paidBy, 0.0) + expense.amount
    }
    return balances
}

Interview Explanation:

  • Backend calculates balances to avoid inconsistency
  • Single source of truth
  • Prevents manipulation on client side
